# First define our utility funciton

cobb_douglas  <- function(x,y) {
  
  x^.3*y^.7
  
}

# Now define a U-max problem

Umax <- function(x1) {
  
  # x cannot be less than 0 or cost more than our whole budget
  x <- min(max(x1,0), m/px)
  
  # y is purchased with remainder of budget 
  y <- (m - px*x)/py 
  
  # display the quantity of x,y chosen
  print(paste0("x:", x, " y:", y))
  
  # optim() minimizes so define function as negative  
  -cobb_douglas(x,y)
  
}

# define prices and income
px <- 2
py <- 5
m  <- 100

#optimize

Part 2: Linear Programming Problem

Problem 1.

\[ \begin{align} & \underset{x_1,x_2}{\text{Max}} & & \pi = 40x_1 + 45x_2 & & \text{s.t.} & & \begin{array}{cc} x_1 + x_2 \leq 600 \\ x_1 \geq 0 \\ x_2 \geq 0 \end{array} \end{align} \]

Without more constraints, the producer would plant 600 acres of hemp (\(x_2\)) because it generates more profit per acre than corn (\(x_1\)).

Problem 2.

\[ \begin{align} & \underset{x_1,x_2}{\text{Max}} & & \pi = 40x_1 + 45x_2 & & \text{s.t.} & & \begin{array}{cc} x_1 + x_2 \leq 600 \\ x_1 + 1.5x_2 \leq 750 \\ x_2 \leq 400 \\ x_1 \geq 0 \\ x_2 \geq 0 \end{array} \end{align} \]

Problem 3.

\[\begin{align} \underbrace{\underset{x_1,x_2}{\text{Max}}}_\text{D.V.} & & \underbrace{\left[\begin{array}{cc} 40 & 45 \end{array}\right] \left[\begin{array}{cc} x_1 \\ x_2 \end{array}\right]}_\text{Objective Function} & & \text{s.t.} & & \underbrace{\left[\begin{array}{cc} -1 & 0 \\ 0 & -1 \\ 0 & 1 \\ 1 & 1.5 \\ 1 & 1 \end{array}\right] \left[\begin{array}{cc} x_1 \\ x_2 \end{array}\right] \leq \left[\begin{array}{cc} 0 \\ 0 \\ 400 \\ 750 \\ 600 \end{array}\right]}_\text{Constraints} \end{align}\]
